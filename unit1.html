sl <- lapply(unique(d$Number), function(X) {
    dd <- d[which(d$Number == X), c("SLon", "SLat", "ELon", "ELat")]


    L <- lapply(1:nrow(dd), function(i) {
        Line(matrix(as.numeric(dd[i, ]), ncol = 2, byrow = TRUE))
    })
    Lines(L, ID = as.character(X))
})
